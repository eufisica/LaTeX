		\question[16] A tensão do cabo que segura o EM-Bird faz um ângulo de \SI{60}{\degree} com a horizontal.\\
		Determine o módulo da resistência do ar, considerando que foi alcançada a velocidade terminal.\\
		Apresente todos os cálculos efetuados. 
		
		\begin{solution}
			\textcolor{white}{.}\\
			\mybox{
				Quando é alcançada a velocidade terminal, o EM-Bird passa a movimentar-se com velocidade constante. Logo, $\vec{F_\text{R}} = 0$. Assim,
				
				\begin{equation*}
					\Bigg \{
					\begin{matrix}
						T_x - R_\text{ar} = 0\\
						T_y - P = 0
					\end{matrix}
					\Leftrightarrow
					\Bigg \{
					\begin{matrix}
						R_\text{ar} = T_x\\
						T_y = mg
					\end{matrix}
					\Leftrightarrow
					\Bigg \{
					\begin{matrix}
						R_\text{ar} = \dfrac{T_y}{\tan(60)}\\
						T_y = 100 \times 9,81
					\end{matrix}
					\Leftrightarrow
					\Bigg \{
					\begin{matrix}
						R_\text{ar} = \SI{566}{\newton}\\
						T_y = \SI{981}{\newton}
					\end{matrix}
				\end{equation*}
				
				A resistência do ar é de \SI{566}{\newton}
			}
		\end{solution}

		\question[16] Os satélites da constelação Iridium descrevem órbitas circulares em Torno da Terra a uma altitude de \SI{780}{\kilo\meter} o que permite estabelecer comunicações em tempo real.\\
		Verifique se estes satélites são geostacionários.\\
		$$R_T = \SI{6400}{km}; \hspace{15pt} m_T = \SI{5,97e24}{\kilogram}$$
		Apresente todas as etapas de resolução.\\
		
		\begin{solution}
			\textcolor{white}{.}\\
			\mybox{
				
				Determinação do raio da órbita do satélite:\\
				
				$r = R_t + h \Rightarrow r = 6400 + 780$\\
				$r = \SI{7280}{\kilo\meter}=\SI{7,28e6}{\meter}$\\
				
				Determinação do período orbital:\\
				$F_c = F_g \Leftrightarrow m a_c = G \dfrac{m_T m}{r^2} \Leftrightarrow$\\
				$\Leftrightarrow \dfrac{v^2}{r} = G \dfrac{m_T}{r^2} \Leftrightarrow v^2 = G \dfrac{m_T}{r} \Leftrightarrow$\\
				$\Leftrightarrow \omega^2 r^2 = G \dfrac{m_T}{r} \Leftrightarrow \dfrac{4\pi^2}{T^2} = G \dfrac{m_T}{r^3} \Rightarrow$
				$\Rightarrow T = \sqrt{\dfrac{4\pi^2r^3}{G m_T}} \Rightarrow T = \sqrt{\dfrac{4\pi^2 \times (7,28 \times 10 ^6)^3}{6,67 \times 10^{-11} \times 5,97 \times 10^{24}}} \Leftrightarrow$
				$\Leftrightarrow T = \SI{6184,82}{\second}$
				
				Determinação do período em horas:\\
				$T = 6184,82 \cancel{\unit{\second}} \times \dfrac{1 \cancel{\unit{\minute}}}{60 \cancel{\unit{\second}}} \times \dfrac{\SI{1}{\hour}}{60 \cancel{\unit{\minute}}} = \SI{1,72}{\hour}$\\
				
				Para ser geoestacionário, o período deveria ser de \SI{24}{\hour} como o resultado é de aproximadamente \SI{2}{\hour}, os satélites não são estacionários.\\	
			}
			
		\end{solution}

		\begin{solution}
			\textcolor{white}{.}\\
			\mybox{
				Aceleração pela equação das velocidades:\\
				$v=v_0 - a t \Rightarrow 0 = v_0 - at \Leftrightarrow a = \dfrac{v_0}{t}$\\
				Equação das posições:\\
				$x=x_0 + v_0 t - \frac{1}{2}at^2 \Leftrightarrow x-x_0 = v_0 t - \frac{1}{2} \cdot \frac{v_0}{t} t^2 \Leftrightarrow$\\
				$\Leftrightarrow 20 = 1,71 t - \frac{1,71}{2} t \Leftrightarrow 20 = 0,85 t \Leftrightarrow t = \SI{23,5}{\second}$
			}
		\end{solution}
