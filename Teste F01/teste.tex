		\question[15] Compare as distâncias percorridas entre o Bernardo Silva e o Jurásek até ao instante em que se cruzam.
		
		\begin{solution}
			\\
			\mybox{
				\begin{minipage}{0.4\linewidth}
					\begin{align*}
						\text{Jogador A}\\
						d_A &= |\Delta x|\\
						&= |-0,4 - 0,6|\\
						&= \SI{1,0}{\meter}
					\end{align*}
				\end{minipage}
				\begin{minipage}{0.4\linewidth}
					\begin{align*}
						\text{Jogador B}\\
						d_B &= |\Delta x|\\
						&= |-0,4 - (-1,0)|\\
						&= \SI{0,6}{\meter}
					\end{align*}
				\end{minipage}
				\\[5pt]
				\begin{minipage}{\linewidth}
					A razão entre as distâncias é
					\begin{equation*}
							\dfrac{d_A}{d_B} = \dfrac{1,0}{0,6} \approx 1,7
					\end{equation*}
				\end{minipage}
			}
		\end{solution}

		\question[15] Mostre que há atrito nas superfícies em contacto no percurso entre B e C.
		\begin{solution}
			\\
			\mybox{
				
					Determinação da aceleração a partir do declive do gráfico entre os instantes \\ $t \epsilon [1,0 ; 1,5] \unit{\second}$ do gráfico:\\
					
					$a = \dfrac{\Delta v}{\Delta t} \Rightarrow a = \dfrac{2,0 - 4,4}{1,5 - 1,0}$\\
					$a = \SI{-4,8}{\meter\per\second\squared}$\\
					
					Determinação da força resultante:\\
					$F_{\text{R}} = m a \Rightarrow F_{\text{R}} = 0,150 \times (-4,8) = \SI{-0,72}{\newton}$\\
					
					Determinação da força resultante devida apenas à componente horizontal do peso (sem força de atrito):\\
					$F_{\text{R}} = -P_x = - m g \sin(\SI{20}{\degree}) \Rightarrow F_{\text{R}} = - 0,150 \times 9,81 \sin(\SI{20}{\degree}) = \SI{-0,50}{\newton}$\\
					
					Logo, no plano inclinado atuou uma força de atrito de intensidade \SI{0,22}{\newton}.\\
					$F_{\text{R}} = -P_x - F_a \Rightarrow -0.72 = -0,50 - F_a \Leftrightarrow F_a = 0,72-0,50 = \SI{0,22}{\newton}$\\
				
			}
			
		\end{solution}

		\question[20] Determine o módulo da aceleração média que atua no atleta entre os instantes $t_0$ e $t_1$. \\Apresente todas as etapas da resolução.
		
		\begin{solution}
			\\
			\mybox{
				Após o jogador perder o contacto com o chão, e admitindo que a resistência do ar é desprezável, sabe-se que para $y_{\text{máx.}} = \SI{0,762}{\meter}$, $v=0$. Assim:
				
				\begin{equation*}
					\Bigg \{
					\begin{matrix}
						0,762 = v_{t_1} t_2 - 4,90 t_2^2\\
						0 = v_{t_1} - 9,8 t_2
					\end{matrix}
					\Leftrightarrow
					\Bigg \{
					\begin{matrix}
						t_2 = \SI{0,394}{\second}\\
						v_{t_1} = \SI{3,86}{\meter\per\second}
					\end{matrix}
				\end{equation*}
				
				Sabendo que entre $t_0$ e $t_1$ decorrem \SI{0,20}{\second}, obtém-se para a aceleração:
				
				\begin{equation*}
					a = \dfrac{\Delta v}{\Delta t} \Rightarrow a = \dfrac{3,86 - 0}{0,20 - 0} = \SI{19,3}{\meter\per\second\squared}
				\end{equation*}
			}
		\end{solution}

			\\
			\mybox{
